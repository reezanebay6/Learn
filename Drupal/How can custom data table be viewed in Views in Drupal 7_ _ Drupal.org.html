fiddly and tricky

Bagz commented 6 years ago

Had to do something similar recently for the revision scheduler module, which has a table that is not described to views. Here is my code, hope it may help you:

/**
 * Implement hook_views_data().
 *
 * Describe table "revision_scheduler" so that it can be used by views
 */
 function mymodule_views_data() {
	$data = array();

	$data['revision_scheduler']['table']['group'] = t('Revision Schedule');
	$data['revision_scheduler']['table']['join'] = array(
		'node' => array(
			'left_table' => 'node_revision',
			'left_field' => 'vid',
			'field' => 'revision_id',
			'type' => 'LEFT',
		),
		'node_revision' => array(
			'left_field' => 'vid',
			'field' => 'revision_id',
			'type' => 'LEFT',
		),
	);
	$data['revision_scheduler']['sid'] = array(
		'title' => t('Revision Schedule ID'),
		'help' => t('Revision Schedule record ID.'),
		'filter' => array(
			'handler' => 'views_handler_filter',
		),
		'field' => array(
			'handler' => 'views_handler_field',
			'click sortable' => TRUE,
		),
		'sort' => array(
			'handler' => 'views_handler_sort',
		),
		'argument' => array(
			'handler' => 'views_handler_argument',
		),
	);
	$data['revision_scheduler']['operation'] = array(
		'title' => t('Scheduled operation'),
		'help' => t('The scheduled moderation operation.'),
		'filter' => array(
			'handler' => 'views_handler_filter_string',
		),
		'field' => array(
			'handler' => 'views_handler_field',
			'click sortable' => TRUE,
		),
		'sort' => array(
			'handler' => 'views_handler_sort',
		),
		'argument' => array(
			'handler' => 'views_handler_argument',
		),
	);
	$data['revision_scheduler']['time_scheduled'] = array(
		'title' => t('Scheduled date'),
		'help' => t('The date that this operation is scheduled for.'),
		'field' => array(
			'handler' => 'views_handler_field_date',
			'click sortable' => TRUE,
		),
		'sort' => array(
			'handler' => 'views_handler_sort_date',
		),
		'filter' => array(
			'handler' => 'views_handler_filter_date',
		),
	);
	$data['revision_scheduler']['time_executed'] = array(
		'title' => t('Executed date'),
		'help' => t('The date that this operation was executed.'),
		'field' => array(
			'handler' => 'views_handler_field_date',
			'click sortable' => TRUE,
		),
		'sort' => array(
			'handler' => 'views_handler_sort_date',
		),
		'filter' => array(
			'handler' => 'views_handler_filter_date',
		),
	);
	return $data;
}

Check out http://drupalcontrib.org/api/drupal/contributions!views!docs!views.api.php/function/hook_views_data/7, it has lots of helpful comments...
